// =========================================================================
// HTTP Snapshot Request Handler (Delivers single frame JPEGs)
// =========================================================================
static esp_err_t snapshot_handler(httpd_req_t *req) {
 camera_fb_t *fb = esp_camera_fb_get();
 if (fb) { esp_camera_fb_return(fb); } // Clear stale buffer references
 fb = esp_camera_fb_get();            // Grab fresh frame from sensor matrices
 if (!fb) {
   httpd_resp_send_500(req);
   return ESP_FAIL;
 }
 httpd_resp_set_type(req, "image/jpeg");
 httpd_resp_set_hdr(req, "Access-Control-Allow-Origin", "*");
 httpd_resp_set_hdr(req, "Cache-Control", "no-cache");
 esp_err_t res = httpd_resp_send(req, (const char *)fb->buf, fb->len);
 esp_camera_fb_return(fb); // Recovers and frees frame memory space back to driver
 return res;
}

// =========================================================================
// HTTP Stream Request Handler (Maintains infinite multipart video chunk loop)
// =========================================================================
static esp_err_t stream_handler(httpd_req_t *req) {
 camera_fb_t *fb = NULL;
 esp_err_t res = ESP_OK;
 size_t _jpg_buf_len = 0;
 uint8_t *_jpg_buf = NULL;
 char part_buf[64];

 res = httpd_resp_set_type(req, STREAM_CONTENT_TYPE);
 if (res != ESP_OK) return res;

 while (true) {
   fb = esp_camera_fb_get();
   if (!fb) {
     res = ESP_FAIL;
   } else if (fb->format != PIXFORMAT_JPEG) {
     // Fallback converter logic if sensor falls out of native JPEG mode
     bool jpeg_converted = frame2jpg(fb, 80, &_jpg_buf, &_jpg_buf_len);
     esp_camera_fb_return(fb);
     fb = NULL;
     if (!jpeg_converted) res = ESP_FAIL;
   } else {
     _jpg_buf_len = fb->len;
     _jpg_buf = fb->buf;
   }

   if (res == ESP_OK) {
     size_t hlen = snprintf(part_buf, 64, STREAM_PART, _jpg_buf_len);
     httpd_resp_send_chunk(req, STREAM_BOUNDARY, strlen(STREAM_BOUNDARY));
     httpd_resp_send_chunk(req, part_buf, hlen);
     httpd_resp_send_chunk(req, (const char *)_jpg_buf, _jpg_buf_len);
   }

   if (fb) {
     esp_camera_fb_return(fb);
     fb = NULL;
   }
   if (res != ESP_OK) break;
 }
 return res;
}

// =========================================================================
// HTTP Web Server Root Handler (Serves control page index)
// =========================================================================
static esp_err_t index_handler(httpd_req_t *req) {
 httpd_resp_set_type(req, "text/html");
 return httpd_resp_send(req, INDEX_HTML, strlen(INDEX_HTML));
}

// =========================================================================
// HTTP Action REST API Parser (Translates query arguments to motor/LED state)
// =========================================================================
static esp_err_t action_handler(httpd_req_t *req) {
 char* buf;
 size_t buf_len;
 char variable[32] = {0,};

 // Unpack URL length parameters
 buf_len = httpd_req_get_url_query_len(req) + 1;
 if (buf_len > 1) {
   buf = (char*)malloc(buf_len);
   if (!buf) {
     httpd_resp_send_500(req);
     return ESP_FAIL;
   }
   // Extract query string data (e.g. "go=forward")
   if (httpd_req_get_url_query_str(req, buf, buf_len) == ESP_OK) {
     if (httpd_query_key_value(buf, "go", variable, sizeof(variable)) != ESP_OK) {
       free(buf);
       httpd_resp_send_404(req);
       return ESP_FAIL;
     }
   } else {
     free(buf);
     httpd_resp_send_404(req);
     return ESP_FAIL;
   }
   free(buf);
 } else {
   httpd_resp_send_404(req);
   return ESP_FAIL;
 }

 // ==========================================
 // Actuator Decision Matrix
 // ==========================================
 if (!strcmp(variable, "forward")) {
   // Right motor value modified with drift deduction to keep trajectory straight
   analogWrite(MOTOR_1_PIN_1, 0);
   analogWrite(MOTOR_1_PIN_2, MAX_SPEED);
   analogWrite(MOTOR_2_PIN_1, 0);
   analogWrite(MOTOR_2_PIN_2, MAX_SPEED - LEFT_DRIFT_COMPENSATION);
 }
 else if (!strcmp(variable, "backward")) {
   analogWrite(MOTOR_1_PIN_1, MAX_SPEED);
   analogWrite(MOTOR_1_PIN_2, 0);
   analogWrite(MOTOR_2_PIN_1, MAX_SPEED - LEFT_DRIFT_COMPENSATION);
   analogWrite(MOTOR_2_PIN_2, 0);
 }
 else if (!strcmp(variable, "left")) {
   // Spin motors in opposing directions using the scaled-down TURN_SPEED constant
   analogWrite(MOTOR_1_PIN_1, 0);
   analogWrite(MOTOR_1_PIN_2, TURN_SPEED);
   analogWrite(MOTOR_2_PIN_1, TURN_SPEED);
   analogWrite(MOTOR_2_PIN_2, 0);
 }
 else if (!strcmp(variable, "right")) {
   // Spin motors in opposing directions using the scaled-down TURN_SPEED constant
   analogWrite(MOTOR_1_PIN_1, TURN_SPEED);
   analogWrite(MOTOR_1_PIN_2, 0);
   analogWrite(MOTOR_2_PIN_1, 0);
   analogWrite(MOTOR_2_PIN_2, TURN_SPEED);
 }
 else if (!strcmp(variable, "stop")) {
   // Safely ground out all driver channels to execute hard brake state
   analogWrite(MOTOR_1_PIN_1, 0);   analogWrite(MOTOR_1_PIN_2, 0);
   analogWrite(MOTOR_2_PIN_1, 0);   analogWrite(MOTOR_2_PIN_2, 0);
 }
 else if (!strcmp(variable, "servo90")) {
   Serial.println("Servo Action: Actuating to 90 degrees");
   analogWrite(SERVO_PIN, 19); // 19 represents approx 2.0ms high time for 90 degrees
 }
 else if (!strcmp(variable, "servo0")) {
   Serial.println("Servo Reset: Returning to original position (0 degrees)");
   analogWrite(SERVO_PIN, 13); // 13 represents approx 1.0ms high time for 0 degrees
 }
 else if (!strcmp(variable, "flashon")) {
   Serial.println("Flash Status: ON via Spacebar");
   digitalWrite(FLASH_LED_PIN, 1);
 }
 else if (!strcmp(variable, "flashoff")) {
   Serial.println("Flash Status: OFF via Spacebar Release");
   digitalWrite(FLASH_LED_PIN, 0);
 }

 httpd_resp_set_hdr(req, "Access-Control-Allow-Origin", "*");
 return httpd_resp_send(req, NULL, 0);
}

// =========================================================================
// Server Initialization (Configures asynchronous dual ports)
// =========================================================================
void startCameraServer() {
 httpd_config_t config = HTTPD_DEFAULT_CONFIG();
 config.server_port = 80; // Main UI control panel port
 if (httpd_start(&camera_httpd, &config) == ESP_OK) {
   httpd_uri_t index_uri = { .uri = "/", .method = HTTP_GET, .handler = index_handler, .user_ctx = NULL };
   httpd_uri_t snap_uri = { .uri = "/snapshot", .method = HTTP_GET, .handler = snapshot_handler, .user_ctx = NULL };
   httpd_uri_t action_uri = { .uri = "/action", .method = HTTP_GET, .handler = action_handler, .user_ctx = NULL };
   httpd_register_uri_handler(camera_httpd, &index_uri);
   httpd_register_uri_handler(camera_httpd, &snap_uri);
   httpd_register_uri_handler(camera_httpd, &action_uri);
   Serial.println("HTTP server started on port 80");
 }

 config.server_port = 81; // Dedicated high-throughput video pipeline port
 if (httpd_start(&stream_httpd, &config) == ESP_OK) {
   httpd_uri_t stream_uri = { .uri = "/stream", .method = HTTP_GET, .handler = stream_handler, .user_ctx = NULL };
   httpd_register_uri_handler(stream_httpd, &stream_uri);
   Serial.println("Stream server started on port 81");
 }
}

// =========================================================================
// Firmware Setup (Initializes GPIO configuration parameters)
// =========================================================================
void setup() {
 // Disable brown-out detector to bypass voltage drop trigger during high current camera/Wi-Fi spikes
 WRITE_PERI_REG(RTC_CNTL_BROWN_OUT_REG, 0);
 Serial.begin(115200);

 // Setup motor controller channel pins as outputs
 pinMode(MOTOR_1_PIN_1, OUTPUT);
 pinMode(MOTOR_1_PIN_2, OUTPUT);
 pinMode(MOTOR_2_PIN_1, OUTPUT);
 pinMode(MOTOR_2_PIN_2, OUTPUT);

 // Initialize Servo Pin Properties
 pinMode(SERVO_PIN, OUTPUT);
 analogWriteFrequency(SERVO_PIN, 50); // Set standard RC Servo PWM frequency to 50Hz
 analogWrite(SERVO_PIN, 13);          // Move servo home to 0 degrees position

 // Initialize Flash LED pin
 pinMode(FLASH_LED_PIN, OUTPUT);
 digitalWrite(FLASH_LED_PIN, 0);

 // Force motors into hard-stop state on startup
 analogWrite(MOTOR_1_PIN_1, 0);   analogWrite(MOTOR_1_PIN_2, 0);
 analogWrite(MOTOR_2_PIN_1, 0);   analogWrite(MOTOR_2_PIN_2, 0);

 // Setup structural interface camera configuration structural definitions
 camera_config_t config;
 config.ledc_channel = LEDC_CHANNEL_0;
 config.ledc_timer = LEDC_TIMER_0;
 config.pin_d0 = Y2_GPIO_NUM;
 config.pin_d1 = Y3_GPIO_NUM;
 config.pin_d2 = Y4_GPIO_NUM;
 config.pin_d3 = Y5_GPIO_NUM;
 config.pin_d4 = Y6_GPIO_NUM;
 config.pin_d5 = Y7_GPIO_NUM;
 config.pin_d6 = Y8_GPIO_NUM;
 config.pin_d7 = Y9_GPIO_NUM;
 config.pin_xclk = XCLK_GPIO_NUM;
 config.pin_pclk = PCLK_GPIO_NUM;
 config.pin_vsync = VSYNC_GPIO_NUM;
 config.pin_href = HREF_GPIO_NUM;
 config.pin_sscb_sda = SIOD_GPIO_NUM;
 config.pin_sscb_scl = SIOC_GPIO_NUM;
 config.pin_pwdn = PWDN_GPIO_NUM;
 config.pin_reset = RESET_GPIO_NUM;
 config.xclk_freq_hz = 20000000;
 config.pixel_format = PIXFORMAT_JPEG;
 config.frame_size = FRAMESIZE_QVGA;  // Resolution size constraint
 config.jpeg_quality = 12;            // Higher compression yields higher frame rate values
 config.fb_count = 2;                 // Dual buffer memory layout allocations

 // Power on camera array module hardware
 if (esp_camera_init(&config) != ESP_OK) { 
   Serial.println("Camera hardware configuration initialization failed!");
   return; 
 }

 // Fine-tune orientation geometry mapping matrices
 sensor_t *s = esp_camera_sensor_get();
 s->set_hmirror(s, 0); 
 s->set_vflip(s, 1);

 // Network link establishment operations
 WiFi.begin(ssid, password);
 WiFi.setSleep(false); // Disables Wi-Fi modem sleep timers to keep stream responses active
 Serial.print("Connecting to WiFi");
 
 while (WiFi.status() != WL_CONNECTED) { 
   delay(500); 
   Serial.print("."); 
 }
 
 Serial.println("\nConnected!");
 Serial.print("IP Address: http://");
 Serial.println(WiFi.localIP());

 startCameraServer();
 Serial.println("Camera control panel running.");
}

// =========================================================================

 // Main worker code is entirely driven by background web socket server callbacks.
 // We use a large structural loop delay block to prevent watchdog timer errors.
 delay(10000);
}
